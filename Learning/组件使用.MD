### ViewPager --> Swiper
```typescript
 private ctl = new SwiperController()
 Swiper(ctl){
  Page1() 
  Page2() 
 }.onChange((index)=>{
   // 切换监听 ，可以用@State修饰的变量实现与其他组件的联动
})
  .loop(false)
  .indicator(false)
  ctl.showNext() // 下一页
  ctl.showPrevious() // 上一页
  ctl.changeIndex(0,true) // 选中
  
 // 用法简单
```

### 列表组件
```typescript
List({space: 10})
{ // 设置ListItem之间的间隔为10
  ListItem()
  { // List的子组件只能是ListItem
    Text('Text1')
      .size({ width: '100%', height: 60 })
      .fontSize(26)
      .backgroundColor('#aabbcc')
  }
  .width('100%')
}
  
//网格列表
Grid() {
  ForEach(this.columns, (item) => { // ForEach语法，循环创建GridItem
    GridItem() {                    // 子组件只能是GirdItem
      Text('Text: ' + item)
        .fontSize(20)
        .backgroundColor('#aabbcc')
        .width('100%')
        .height(70)
    }
  })
}
.columnsTemplate("1fr 1fr 1fr")     // 设置Grid为3列，并且均分
  .columnsGap(10)                     // 设置列间距为10vp
  .rowsGap(10)                        // 设置行间距为10vp
  .width('100%')
  .height(170)
  .backgroundColor(Color.Pink)
```
